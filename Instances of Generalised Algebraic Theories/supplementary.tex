Using the notation just introduced we can give a modified version of lemma \lref{substitutioninterpretationlemma}.

\begin{lemma}
\llabel{supplementarylemma}
If $I$ is an interpretation of the generalised algebraic theory $U$ in a contextual category \catcw then
in the general situation described in lemma \lref{substitutioninterpretationlemma}, the rule
\gatdisplayrule{Q}{\isT{\Omega[s_1|y_1...s_m|y_m]}} 
 is mapped by $I$ to object $\duple{f_1,...f_m}^*b$ and
the rule \gatdisplayrule{Q}{\ofT{s[s_1|y_1...s_m|y_m]}{\Omega[s_1|y_1...s_m|y_m]}} is mapped by $I$ to section   $\duple{f_1,...f_m}^*f$
\end{lemma}

\begin{lemma}
\llabel{supplementarylemma2}
If $I$ is an interpretation of the generalised algebraic theory $U$w in a contextual category \catcw then
in the general situation described in lemma \lref{substitutioninterpretationlemma} \highlight{except} if $s_i$  maps to $s(f_i)$ instead of to $f_i$ then the rule
\gatdisplayrule{Q}{\isT{\Omega[s_1|y_1...s_m|y_m]}} 
 is mapped by $I$ to object $\tuple{f_1,...f_m}^*b$ and
the rule \gatdisplayrule{Q}{\ofT{s[s_1|y_1...s_m|y_m]}{\Omega[s_1|y_1...s_m|y_m]}} is mapped by $I$ to section   $\tuple{f_1,...f_m}^*f$
\end{lemma}
