

\newcommand{\w}[1]{
\psframebox[framesep=1pt]{\strut \textcolor{black}{#1}}
}

\newcommand{\p}[1]{
\psframebox[linewidth=1pt,linecolor=black]{\textcolor{black}{#1}}
}


\newcommand{\prw}[1]{
\psframebox[linewidth=1pt,linecolor=white,border=.5pt]{\textcolor{black}{#1}}
}

\newcommand{\tagp}[2]{
\raisebox{- \height * \real{0.5}}{\raisebox{\height}{\psframebox{\textcolor{black}{#1}}}{\footnotesize{#2}}}
}

\newcommand{\tagw}[2]{
\raisebox{- \height * \real{0.5}}{\raisebox{\height}{\psframebox[framesep=1pt]{\strut \textcolor{black}{#1}}}{\footnotesize{#2}}}
}

\newcommand{\wnode}[1]{
\TR{\w{#1}}
}

\newcommand{\tagwnode}[2]{
\TR{\psframebox[framesep=1pt]{\strut \textcolor{black}{#1}}}\nput[labelsep=0.01]{0}\pssucc{\raisebox{-0.7cm}{\footnotesize{#2}}}
}


\newcommand{\phrnode}[2]{

}
