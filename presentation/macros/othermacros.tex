\newcommand{\nakedbinarysourcediagram}[5]{
\begin{array}{c p{0.5cm} c}
             &&   \Rnode{b}{#2}\\[0.01cm]
\Rnode{a}{#1} &&               \\[0.01cm] 
             &&   \Rnode{c}{#3}
\end{array} 
\begin{arrows}
\ncarr{a}{b}
\alabel{#4}
\ncarr{a}{c}
\blabel{#5}
\end{arrows}
}

\newcommand{\nakedmultisourcediagram}[4]{
\nakedmultisourceobjects{#1}{#2}{#3}{#4}
\begin{arrows}
\ncarr{a}{b1}
\alabel{#4_1}[0.5]
\ncarr{a}{b2}
\alabel{#4_2}[0.5][-1]
\ncarr{a}{bn}
\blabel{#4_{#1}}[0.5][-1]
\end{arrows}
}
