
 Let \Rel be the category of sets and binary relations. 
For every relation $R:x \morph y$ in \Rel 
there is a converse relation $R^{-1} :y \morph x$,
 a restriction idempotent $\overline{R}: x \morph x$ and 
 a range idempotent $\widehat{R} : y \morph y$ such that
$\widehat{R} = \overline{R^{-1}}$.

 Every partial function can be considered as a binary relation and the inclusion of $\Par$ in $\Rel$ is functorial
and preserves the restriction and range idempotent, we shall denote 
by $U$ as it 'forgets' the many-one nature of a partial function so
that
$\Par \xrightarrow{U} \Rel$.

It is simple to check that for any $R:x \morph y$ in \Rel, $R$ is in the image of $U$ and so
$R$ is single valued iff $R^{-1} \circ R = \widehat{R}$. 


If $F: \catc \morph \Par$ is a functor then composition of $F$ yields a functor into $\Rel$ which we shall denote $F_R$ so that 
\commentary{do we actually use this?}
 $F_R : \catc \morph \Rel$.

 \begin{lemma}
 \llabel{threesidedlemma}
 If in the category \Par the following diagram of partial functions commutes 
$
\begin{array}{c p{0.25cm} c  p{0.25cm} c }
             &&   \Rnode{c}{c} &&              \\[0.4cm]
\Rnode{b1}{b_1} &&                   && \Rnode{b2}{b_2} \\[0.4cm]
             &&   \Rnode{a}{a} &&              
\end{array} 
\begin{arrows}
\ncarr{a}{b1}
\alabel{f_1}
\ncarr{a}{b2}
\blabel{f_2}
\ncarr{b1}{c}
\alabel{g_1} 
\ncarr{b2}{c}
\blabel{g_2}
\end{arrows}
$
commutes then  the diagram of relationships
$
\begin{array}{c p{0.25cm} c  p{0.25cm} c }
             &&   \Rnode{c}{c} &&              \\[0.4cm]
\Rnode{b1}{b_1} &&                   && \Rnode{b2}{b_2} \\[0.4cm]
             &&   \Rnode{a}{a} &&              
\end{array} 
\begin{arrows}
\ncarr{b1}{a}
\blabel{U(f_1)^{-1}}
\ncarr{a}{b2}
\blabel{U(f_2)}
\ncarr{b1}{c}
\alabel{U(g_1)} 
\ncarr{b2}{c}
\blabel{U(g_2)}
\end{arrows}
$
commutes in the category \Rel.
 \end{lemma}
 \begin{proof}
 Because $U(f_1)$ is single valued we have
 \begin{align*}
 U(f_1)^{-1} \circ U(f_2) \circ U(g_2)
    &=  U(f_1)^{-1} \circ U(f_2 \circ g_2) \\
    &= U(f_1)^{-1} \circ U(f_1 \circ g_1) \\
    &= U(f_1)^{-1} \circ U(f_1) \circ U(g_1) \\
    & = U(g_1).
 \end{align*}
 \end{proof}

Providing that we assume the axiom of choice then
for any $R:x \morph y$ in \Rel there is a partial function
$f:x \morph y$ such that $x \leq y$ and $bar{f} = \bar{R}$.
We need the following variation on this 
\begin{lemma}
 \llabel{atleasttwopartialfunctionslemma}
 If $R: x \morph y$ in the category \Rel and $q:y \morph z$ and $f: x \morph y$ in the category \Par such that $\overline{R} \leq \bar{q}$ and $R \circ q = f$ then 
 if $R$ is not single-valued
 then there exists at least  two distinct partial functions $r: x \morph y$ 
 such that $r \circ q = f$.
 \end{lemma}
 \begin{proof}

We can choose $r:x \morph y$ such that $r \leq R$ and $\bar{r} = \bar{R}$.  $r$ is bound to satisy the condition $r \circ q =f$
as we now show. 

First note that because $R \circ q =f$ and because $r \leq R$
then $r \circ q \leq f$. Then we have 
\begin{align*}
r \circ q 
   &= \overline{r \circ q} \circ f 
          &&\mbox{because $r \circ q \leq f$, } \\
   &= \overline{r} \circ f
          &&\mbox{because $\hat{r} \leq \bar{q}$,} \\
   &= \overline{R} \circ f
     &&\mbox{because $r$ chosen so that $\bar{r} = \bar{R}$,} \\
   &= \overline{R \circ q} \circ f
     &&\mbox{because it is given that $\bar{R} \circ q = \bar{R}$}, \\
   &= \bar{f} \circ f 
          &&\mbox{because it is given that $R \circ q = f$,} \\
   &= f
          &&\mbox{by R.1.} 
\end{align*}

Now there is an $a \in x$ such that $R(x)$ has more than a single value and therefore a value $b$ which is distonct from $r(x)$. Define a second function $r'$ to be identical to
$r$ except at $x$ where we define its value to be $b$.
$r':x \morph y$ is distinct from $r$ and satisfies  
$r \leq R$ and $\bar{r} = \bar{R}$. Therefore as before 
we have $r' \circ q = f$.

\end{proof}


 \begin{lemma}
 \llabel{functionalBackupLemma}
 If $R_1: x \morph y$ and $R_2: y \morph z$ in the category \Rel
 then if $R_2$ is defined at every element $b$ of $y$ which is in the image of $R_1$,
 then if $R_1 \circ R_2$ is single valued then $R_2$ is single valued. \footnote{Note that if we defined restriction operators and range operators for morphisms of \Rel,
 which we could do, then we could write the key assumption more succinctly as
 $widehat{R_1} = \overline{R_2}$.} 
 \end{lemma}
 \begin{proof}
 Staightforward. If $b_1$ and $b_2$ are elements of $y$ 
 for which $R_2$ is defined and for which $R_2(b_1) = R_2(b_2)$
  then since $b_1$ must also be in the range of $R_1$ then there are $a_1$ and $a_2$ such that $R_1(a_1)=b_1$
 and $R_2(a_2)=b_2$. Now 
 $(R_1 \circ R_2) (a_1)
 =R_2(b_1) = R_2(b_2) 
 = (R1 \circ R_2) (a_2)$ therefore $a_1 = a_2$ since $R_1 \circ R_2$ is single-valued. Therefore $b_1=R_1(a_1) = R_2(b_2)$.
 \end{proof}
