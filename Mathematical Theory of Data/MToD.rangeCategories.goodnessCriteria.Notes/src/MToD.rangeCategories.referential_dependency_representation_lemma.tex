\subsection{Lemma 2C  --- Representation of Referential Inclusion Dependencies}
\begin{worktt}
\highlight{COPIED FROM preparation}
\begin{lemma}
\label{catincdsrepresented}
If $\catc$ is a locally finite range category and $\reqtc$ is a set of instances, if $\catc$ is 
\term{maximally constrained} to the requirement $\reqtc$ then
every referential inclusion dependency with respect to $\reqtc$ is represented in $\catc$.
\end{lemma} 
\begin{proof}
Suppose such  a range category $\catc$ and  requirement $\reqtc$ 
 and suppose there is a referential inclusion dependency
$a[f_1,...f_n] \overset{R}{\subseteq} b[q_1,..q_n]$ with respect to $\reqtc$,
where
$
\begin{array}{c p{0.25cm} c  p{0.25cm} c }
             &&   \Rnode{b1}{b_1} &&              \\[0.4cm]
\Rnode{a}{a} &&                   && \Rnode{c}{c} \\[0.4cm]
             &&   \Rnode{bn}{b_n} &&              
\end{array} 
$
\ncarr{a}{b1}
\alabel{f_1}
\ncarr{c}{b1}
\blabel{q_1} 
\ncarr{a}{bn}
\blabel{f_n}
\ncarr{c}{bn}
\alabel{q_n}
in $\catc$. 

\newcommand{\barffn}{\overline{f_1} \circ ... \circ \overline{f_n}}

From sketch $S$ of $C$ we can construct a sketch $S'$ by formally adding a morphism $\qq{r}: a \morph c$
and path equivalences, 
for each $i$, $1 \leq i \leq n$, $\qq{r} \circ q_i = \overline{\qq{r}} \circ f_i$ and
$\overline{\qq{r}}=\barffn$.
Let $\catcp$ be the range category generated by $S'$ and
let $I: C \morph C'$ be the inclusion functor generated 
by the inclusion of $S$ in $S'$. 
$I$ is consistent with $\reqtc$ since
each instance $D \in  \reqtc$ can be uniquely extended to a functor $D': \catcp \morph \Par$ by defining $D'(\qq{r})=R_D$.
\commentary{go infinite - avoid stcikleback category having to be locally finite!!}
Since $\catc$ is a range category then there is an
embedding $E$ of $C$ into the short stickleback category $\Stk$
and a functor $\HomStk: \Stk \morph \Par$
such that the composition $E \circ \HomStk$ is a range functor.

Since \catcw is maximally constrained and because $I$ is consistent with $\reqtc$ then the functor 

$E \circ \HomStk$  can be extended 
to a functor $H_a: C' \morph \Fin$ so that the following diagram of functors commutes 
\[
\begin{array}{c p{0.3cm} c p{0.3cm} c}
                 && \Rnode{t}{\Stk}   &&                     \\[0.7cm]
\Rnode{l}{\catc} &&                   &&     \Rnode{r}{\Par} \\[0.7cm]
                 && \Rnode{b}{\catcp} &&
\end{array}
\begin{arrows}
\ncarr{l}{t}
\alabel{E}
\ncarr{t}{r}
\alabel{\HomStk}[0.75]
\ncarr{l}{b}
\blabel{I}
\ncarr{b}{r}
\blabel{H_a}
\end{arrows}
\] 

In particular we must have 

$H_a(a)=\HomStk[a][a]$,

$H_a(c)=\HomStk[a][c]$ and thus 

$H_a(\qq{r}):\HomStk[a][a] \morph \HomStk[a][c]$ which 

applying to $E(\overline{\qq{r}})$ which, by the way, is the singleton stickleback
$\tuple{\barffn}$ gives us

$H_a(\qq{r})(E(\overline{\qq{r}})): a \morph c$ in $\Stk$. 


For each $i$, from this commuting diagram in \catcp
\[
\upTriangle{b_i}{a}{c}{\qq{r}}{f_i}{q_i}
\]
applying functor $H_a$
we infer this diagram commutes in \Par
\[
\upTriangle{\HomStk[a][a]}
            {\HomStk[a][b_i]}
            {\HomStk[a][c]}
            [1.5cm]           % row spacing (height adjustment)
            {H_a(\qq{r})}
            {\HomStk[a][\tuple{f_i}]}
            {\HomStk[a][\tuple{q_i}]}
            [0.2].     % npos for labels on sides of triangle 
\]

Now since there is singleton stickleback $\tuple{\barffn}: a \morph a$ in 
  $\Stk$ the commutivity of this diagram gives
\[
H_a(\qq{r}) (\tuple{\barffn}) \circ \tuple{q_i} 
= \tuple{\barffn} \circ \tuple{f_i}
= \tuple{\barffn \circ f_i} 
\] 
in $\Stk$.

Suppose $H_a(\qq{r}) (\tuple{\barffn})$ 
is a short stickleback $G =\tuple{g_1,etc, g_m}$ of length m. 
Then LHS above is $\tuple{g_1,etc, g_m \circ q_i}$.

\commentary{I am assuming $q_i$ is total but it may be possible toavoid this assumption.}
From an as yet unwritten sequential shortening lemma we can deduce that there are
(shortening) morphisms $\sm[-1]$ so that, \foreachj[m-2],
the domain of  $s_j$ is the codomain of $g_j$ and 
the codomain  is $b_i$ such that 
\[
g_1 \circ s_1 = f_i
\]
and \foreachj[m-1]
\[
g'_j \circ s_j = g_{j+1} \circ s_{j+1}
\]
and
\[
g'_{m-1} \circ s_{m-1} = g_m \circ q_i
\]

In any instance $D \in \reqtc$ therefore we get
\[
D(g_1) \circ D(s_1) = D(f_i)
\]
and \foreachj[m-2]
\[
D(g'_j) \circ D(s_j) = D(g_{j+1}) \circ D(s_{j+1})
\]
and
\[
D(g'_{m-1}) \circ D(s_{m-1}) = D(g_m) \circ D(q_i)
\]

From the latter two, and by use of  lemma \lref{threesidedlemma} we get
\[
D(s_j) = D(g'_j) \circ D(g_{j+1}) \circ D(s_{j+1})
\]
and
\[
 D(s_{m-1}) = D(g'_{m-1}) D(g_m) \circ D(q_i)
\]
this enables us to iteratively remove instance of $s_j$ from the identity (first of last three)
to obtain 
\[
D(f_i) = D(g_1) \circ D(g'_1)^{-1} \circ ... D(g'{m-1})^{-1} \circ D(g_m) \circ D(q_i) 
\]

Thus $D(g_1) \circ D(g'_1)^-1 \circ ... D(g'{m-1}^{-1} \circ D(g_m)$
is a relation, call it $G_D$, in \Rel such that $G_D \circ q_i = f_i$, \foreachi.
But there is a unique single valued relationship $R_D$ such that 
$R_D \circ q_i = f_i$. Therefore by lemma \lref{???} we have that $G_D$ is single-valued. We have that in each $D$
\[
D(g_1) \circ D(g'_1)^-1 \circ ... D(g'{m-1}^{-1} \circ D(g_m) = R_D
\]
and so we have shown that the referential inclusion dependency $R$ isrepresented 
in \catcw by a stickleback $G$. It remains to show that $G$ is in fact
a single morphsim. This follows from an as yet unwritten corollary
to the lemma regarding representation of functional dependencies.
\end{proof}

\end{worktt}
