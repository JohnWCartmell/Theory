DOCUMENT HOW TO USE VARIOUS ZIGZAG COMMANDS

FIRST CHECK OUT fancyvrb so as not to repeat example code


\hrule

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagZeroOne{x}{y}{f}
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagZeroOne{x}{y}{f}[g]
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagZeroOneShort{x}{y}{f}
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagZeroOneShort{x}{y}{f}[g]
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagStart{x}{y}{f}{n}
\]
\end{showexample}



\begin{showexample}
\[
\arraycolsep=7pt
\zigzagDotsThenFinal{x}{y}{f}{n}
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagDotsThenFinalFlex{x}{y}{f}{n}{m}
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzag{x}{y}{f}{n}
\]
\end{showexample}


\begin{showexample}
\[
\arraycolsep=7pt
\zigzag{x}{y}{f}[g]{n}
\]
\end{showexample}

\begin{showexample}
\[
\arraycolsep=7pt
\zigzagFlex{x}{y}{f}{n}{n+m}
\]
\end{showexample}
