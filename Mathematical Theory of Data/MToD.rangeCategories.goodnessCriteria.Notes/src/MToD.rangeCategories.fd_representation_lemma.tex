
\subsection{Lemma 2B --- Representation of Functional Dependencies}

\begin{lemma}
\llabel{functionaldependencyrepresentationlemma}
If $\catc$ is a locally finite \newt{RR.5 range} category and $\reqtc$ is a H of instances, if $\catc$ is 
\term{maximally constrained} to the requirement $\reqtc$ then
all functional dependencies $\fundep{H}{f}{g}$  with respect to $\reqtc$ are represented in $\catc$.
\end{lemma}

\begin{lemma}
If $Z$ is a tight zigzag in a range category \catcw which is maximally constrained to a requirement \reqt, if for every $D \in \reqt$, $D^*(Z)$ is single valued
then either $Z$ is or can be shortened to a singleton stickleback.
\end{lemma}
\begin{proof}
If $Z$ is not already of length $1$, if it is a  
zigzag  $\zigzagTuple{z}{m}$, for some 
$m > 1$ then
since in every $D \in \reqt$ the relationship 
$D^*(Z)=D(\zigzagTuple{z}{m-1}) 
    \circ D(z_{m-1})^{-1} \circ D(z_m)$
is single valued then by lemma \lref{functionalBackupLemma} then
relationship $D(z_{m-1})^{-1} \circ D(z_m)$ is
single valued i.e. there is a functional dependency 
from $z_{m-1}$ to $z_m$ in \catcw wrt requirement
\reqt. Therefore by our previous 
lemma, \ref{functionaldependencyrepresentationlemma}, there is a morphism $g$ in \catcw such that
$z_{m-1} \circ z = z_m$. So $Z$ shortens. This process can be continued until $Z$ is of length 1. 
\end{proof}

\begin{corollary}
\llabel{functionalDependencyRepresentationCorollary}
If $Z$ is a short 
stickleback  of a range category \catcw which is maximally constrained to a requirement \reqt, if for every $D \in \reqt$, $D^*(Z)$ is single valued, then
$Z$ is of length $1$ i.e. $Z=\tuple{z}$ for some morphism $z$ of \catcw.
\end{corollary}
